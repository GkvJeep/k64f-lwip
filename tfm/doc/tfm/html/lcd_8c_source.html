46 void LCD_Init(uint8_t lcd_addr, uint8_t lcd_cols, uint8_t lcd_rows, uint8_t charsize)
47 {
48  g_lcd_addr = lcd_addr;
49  g_lcd_cols = lcd_cols;
50  g_lcd_rows = lcd_rows;
51  g_lcd_charsize = charsize;
52  g_lcd_backlightval = LCD_BACKLIGHT;
53 
54  select_slave(g_lcd_addr);
55 
56  /* Set bit mode, number of rows, and font size. */
57  g_lcd_displayfunction = LCD_4BITMODE | LCD_2LINE | LCD_5x8DOTS;
58 
59  /* For some 1 line displays you can select a 10 pixel high font. */
60  if ((g_lcd_charsize != 0) && (g_lcd_rows == 1))
61  {
62  g_lcd_displayfunction |= LCD_5x10DOTS;
63  }
64 
65  /* Issue the setting command. */
66  LCD_command(LCD_FUNCTIONSET | g_lcd_displayfunction);
67 
68  /* Set the display on, no cursor, nor blinking by default. */
69  g_lcd_displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
70  LCD_command(LCD_ENTRYMODESET | g_lcd_displayfunction);
71 
72  /* Set the text direction. */
73  g_lcd_displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
74 
75  /* Go home. */
76  LCD_home();
77 }
